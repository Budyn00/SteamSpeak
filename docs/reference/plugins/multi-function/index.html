Multi Function

This plugin allows you to display relevant server information on the configured channels.

#Configuration

module.exports.info = {
  name: 'Multi function',
  description:
    'This plugin allows you to display relevant server information on the configured channels.',
  config: {
    enabled: true,
    data: [
      {
        enabled: true,
        channelId: 18,
        name: '» Server ping: [PING]ms'
      },
      {
        enabled: true,
        channelId: 21,
        name: '» Server packet loss: [PACKETLOSS]%'
      },
      {
        enabled: true,
        channelId: 16,
        name: '» Channels count: [CHANNELS]'
      },
      {
        enabled: true,
        channelId: 24,
        name: '» Bytes uploaded: [UPLOAD]'
      },
      {
        enabled: true,
        channelId: 25,
        name: '» Bytes downloaded: [DOWNLOAD]'
      }
    ],
    interval: {
      weeks: 0,
      days: 0,
      hours: 0,
      minutes: 0,
      seconds: 5
    }
  }
};
